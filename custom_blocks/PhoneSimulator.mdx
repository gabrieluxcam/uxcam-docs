import React from 'react';

export const PhoneSimulator = ({
  width = '300px',
  height = '600px',
  frameColor = '#000',
  backgroundColor = '#fff',
  notch = true,
  children,
}) => {
 return (
    <div

      style={{

        height,

      }}
      >
      {/* Notch */}
      {notch && (
        <div
          style={{

            left: '50%',

          backgroundColor,
          boxShadow: 'inset 0 0 6px rgba(0,0,0,0.1)',

  );
};
